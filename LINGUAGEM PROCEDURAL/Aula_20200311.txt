1)Crie uma procedure que leia dois números quaisquer e informa se o primeiro é maior, igual ou menor do que o segundo.

delimiter $$
create procedure exercicio_1 (in numero1 int, in numero2 int, out status varchar(50))
begin 
	if (numero1 > numero2) then
		set status = "Primeiro é maior que o segundo.";
	elseif (numero1 = numero2) then
		set status = "Primeiro é igual ao segundo.";
	else
		set status = "Primeiro é menor que o segundo";    
	end if;
end;

set @resultado = "";
call exercicio_1(2, 4, @resultado);
select @resultado;

    declare maior int;

5)Crie uma procedure que leia a renda de uma pessoa e calcula um imposto, com base na tabela de alíquotas abaixo. 
Até 	1.200,00	isento
de 	1.201,00 a 5.000,00	10%
de	5.001,00 a 10.000,00	15%
acima de 10.000,00		20%

delimiter $$
create procedure exercicio_5 (in renda decimal(10,2), out imposto decimal(10,2))
begin 
	if (renda <= 1000) then
	  set imposto = 0;
	elseif renda >= 1201 and renda <= 5000 then
      set imposto = renda * 0.10;
	elseif renda >= 5001 and renda <= 10000 then
      set imposto = renda * 0.15;
	else
      set imposto = renda * 0.20;
	end if;
end;

set @imposto = 0;
call exercicio_5(5000,@imposto);
select @imposto;
